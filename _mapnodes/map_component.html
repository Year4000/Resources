This component is required, this tells MapNodes information about the map. It includes a name, version, description, authors. 

"map": {
    // the name of the map
    "name": "The Map", 

    // the version of the map
    "version": "1.0",

    // the goal or description of the map
    "description": "This is the map where you kill the other team.",

    // the list of authors for the map, place your username here
    "authors": ["Author 1", "Author 2"]
}
