Teams are not only teams but groups. This controls how teams work. 

"teams": {
    // the id of the team, must match name but lowercase
    "": {
        // the display name of the team, can be any case
        "name": "",

        // the color of the team, use ChatColor.
        "color": "",

        // the id of the kit you made above
        "kit": "",

        // the size of the team defaults to 1
        "size": 1,

        // the spawns that this team will randomly spawn at, you may use `point` and/or `min/max`
        // take a look at Twin Tower for square spawns.
        // take a look at Desert Fort for multi spawns.
        "spawns": [
            {"point": {"x": 0, "y": 0, "z": 0}},
            {"min":{"x":-0,"y":0,"z":-0},"max":{"x":0,"y":0,"z":0}}
        ]
    }
}
